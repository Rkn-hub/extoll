function doPost(e) {
  try {
    // Open your spreadsheet by ID
    // Find your spreadsheet ID in the URL: https://docs.google.com/spreadsheets/d/SPREADSHEET_ID/edit
    var spreadsheetId = "PASTE_YOUR_SPREADSHEET_ID_HERE";
    var sheet = SpreadsheetApp.openById(spreadsheetId).getSheets()[0];
    
    // Get form data
    var fullName = e.parameter.fullName || '';
    var email = e.parameter.email || '';
    var phone = e.parameter.phone || '';
    var message = e.parameter.message || '';
    var timestamp = new Date();
    
    // Add to spreadsheet (matches your Google Form columns)
    sheet.appendRow([timestamp, fullName, email, phone, message]);
    
    // Return success
    return ContentService
      .createTextOutput(JSON.stringify({'result': 'success'}))
      .setMimeType(ContentService.MimeType.JSON);
      
  } catch (error) {
    return ContentService
      .createTextOutput(JSON.stringify({'result': 'error', 'error': error.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  return ContentService.createTextOutput("Script is working!");
}
